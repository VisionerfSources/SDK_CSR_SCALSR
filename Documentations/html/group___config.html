This function is used to get the list of configuration datas.

Parameters
[in]pCirrus3DHandler: Data previously initialized with the VN_Cirrus3DHandler_initialize() function
[out]pNbParamInList: Number of data available
[out]DataList: List of data available
Return values
VN_eERR_NoErrorif success
Errorcode otherwise

Example :

// Get the list of data available in the Cirrus3DSensor then get the data informations
{
VN_tERR_Code err = VN_eERR_NoError;
VN_tDataRoot dataList[VN_cSizeList];
int nbData=0;
VN_tDataInt32 dataInt32;
VN_tDataReal32 dataReal32;
VN_tDataString dataString;
err=VN_Cirrus3DHandler_GetDataRootList(&Cirrus3DHandler,
&nbData,
dataList);
if (err!=VN_eERR_NoError) return err;
for(int i=0; i<nbData; i++)
{
switch (dataList[i].paramtype) {
case VN_eTypeInt32:
{
err=VN_Cirrus3DHandler_GetDataInt32(&Cirrus3DHandler,
dataList[i].paramId,
&dataInt32);
if (err!=VN_eERR_NoError) return err;
printf("Name %s - ",dataInt32.dataInfo.name);
printf("Value %d\n",dataInt32.value);
}
break;
case VN_eTypeReal32:
{
err=VN_Cirrus3DHandler_GetDataReal32(&Cirrus3DHandler,
dataList[i].paramId,
&dataReal32);
if (err!=VN_eERR_NoError) return err;
printf("Name %s - ",dataReal32.dataInfo.name);
printf("Value %f\n",dataReal32.value);
}
break;
case VN_eTypeString:
{
err=VN_Cirrus3DHandler_GetDataString(&Cirrus3DHandler,
dataList[i].paramId,
&dataString);
if (err!=VN_eERR_NoError) return err;
printf("Name %s -",dataString.dataInfo.name);
printf("Value %s\n",dataString.value);
}
break;
default:
return VN_eERR_UnknownData;
break;
}
}
}
